library(shiny)
ui <- fluidPage(
  titlePanel("Affordable Laptop Finder"),
  numericInput("Budget", "Enter your Budget in GBP - Enter 5500 If No Budget", value = 5500, min = 0, step = 100),
  selectInput("sortOrder", "Sort by Price", choices = c("Ascending" = "asc", "Descending" = "desc")),
  tableOutput("affordablelaptop")
)

server <- function(input, output) {
  output$affordablelaptop <- renderTable({
    affordable <- Laptopdf[Laptopdf$Price_GBP <= input$Budget, ]
    
    if (input$sortOrder == "asc") {
      affordable <- affordable[order(affordable$Price_GBP), ]
    } else {
      affordable <- affordable[order(affordable$Price_GBP, decreasing = TRUE), ]
    }
    
    affordable[, c("Company", "Product", "Price_GBP")]
  })
}

shinyApp(ui = ui, server = server)
